# Define a function to calculate the salary in USD
def calculate_salary_usd(comp_dict):
    if not isinstance(comp_dict, dict):
        return None
    
    interval = comp_dict.get('interval')
    min_amount = comp_dict.get('min_amount')
    max_amount = comp_dict.get('max_amount')
    
    # Calculate average if both min and max are available
    if min_amount is not None and max_amount is not None:
        avg_amount = (min_amount + max_amount) / 2
    # Use min_amount if max_amount is not available
    elif min_amount is not None:
        avg_amount = min_amount
    # Use max_amount if min_amount is not available
    elif max_amount is not None:
        avg_amount = max_amount
    else:
        return None
    
    # Convert based on interval
    if interval == 'hourly':
        return avg_amount * 2000  # Convert hourly to yearly (2000 hours)
    elif interval == 'yearly':
        return avg_amount
    else:
        # Handle other intervals if needed (monthly, weekly, etc.)
        return None
