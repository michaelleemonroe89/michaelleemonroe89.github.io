# Create combined dataset
month_by_city <- merged_data %>%
  filter(!is.na(Start.Month)) %>%
  group_by(City, Start.Month) %>%
  summarise(count = n(), .groups = 'drop') %>%
  mutate(plot_type = "By City")

month_combined <- merged_data %>%
  filter(!is.na(Start.Month)) %>%
  group_by(Start.Month) %>%
  summarise(count = n(), .groups = 'drop') %>%
  mutate(City = "All Cities", plot_type = "Combined")

# Combine datasets
month_all <- bind_rows(month_by_city, month_combined)

# Plot it
ggplot(month_all, aes(x = Start.Month, y = count, fill = City)) +
  geom_col(position = "dodge") +
  facet_wrap(~plot_type, scales = "free") +
  labs(
    title = "Most Popular Months for Bike Trips",
    x = "Month",
    y = "Number of Trips",
    fill = "City"
  ) +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

# Get the #1 trip for each city
trips_by_city <- merged_data %>%
  group_by(City, Start.Station, End.Station) %>%
  summarise(count = n(), .groups = 'drop') %>%
  group_by(City) %>%
  arrange(desc(count)) %>%
  slice(1) %>%
  mutate(trip_route = paste0(substr(Start.Station, 1, 20), "  ", substr(End.Station, 1, 20)))
# Plot it
ggplot(trips_by_city, aes(x = reorder(trip_route, count), y = count, fill = City)) +
  geom_col() +
  coord_flip() +
  labs(
    title = "Most Popular Trip Route by City",
    x = "Trip Route (Start  End)",
    y = "Number of Trips",
    fill = "City"
  ) +
  geom_text(aes(label = count), hjust = 2) +
  theme_minimal()

# Birthyear info
birth_year_info <- merged_data %>%
  filter(City %in% c('New York', 'Chicago') & !is.na(Birth.Year)) %>%
  group_by(City) %>%
  summarise(
    "Earliest Birth Year" = min(Birth.Year),
    "Recent Birth Year" = max(Birth.Year),
    "Most Common Birth Year" = as.numeric(names(sort(table(Birth.Year), decreasing = TRUE))[1]),
